242 case 'x': {
243 float newY = y * cosA - z * sinA;
244 float newZ = y * sinA + z * cosA;
245 y = newY;
246 z = newZ;

248 }
249 case 'y': {
250 float newX = x * cosA + z * sinA;
251 float newZ = -x * sinA + z * cosA;
252 x = newX;
253 z = newZ;
254 break;
255 }
256 case 'z': {
257 float newX = x * cosA - y * sinA;
258 float newY = x * sinA + y * cosA;
259 x = newX;
260 y = newY;
261 break;
262 }
263 default:
